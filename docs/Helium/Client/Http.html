def paginated_get(path, opts = {})
  klass  = opts.fetch(:klass)
  params = opts.fetch(:params, {})

  Cursor.new(client: self, path: path, klass: klass, params: params)
end

def url_for(path)
  return path if path =~ /^http/

  path = path.gsub(/^\//, '')
  "#{base_url}/#{path}"
end
